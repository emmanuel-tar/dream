EMAILJS_PUBLIC_KEY = 'YOUR_PUBLIC_KEY'EMAILJS_SERVICE_ID = 'YOUR_SERVICE_ID'EMAILJS_TEMPLATE_COMPANY = 'ds_company_alert'
